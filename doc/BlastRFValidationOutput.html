def validation

  # if there are different reading frames of the same sign
  # count for positive reading frames
  count_p = 0
  count_n = 0
  frames_histo.each do |x, y|
    if x > 0
      count_p += 1
    else
      if x < 0
        count_n += 1
      end
    end
  end

  if count_p > 1 or count_n > 1
    :no
  else
    :yes
  end

end
